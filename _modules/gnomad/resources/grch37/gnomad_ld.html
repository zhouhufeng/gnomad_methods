from gnomad.resources.resource_utils import TableResource, BlockMatrixResource
from gnomad.resources.grch37.gnomad import CURRENT_EXOME_RELEASE, CURRENT_GENOME_RELEASE
from typing import Optional



def _ld_matrix_path(data_type: str, pop: str, common_only: bool = True, adj: bool = True, version: Optional[str] = None):
    if version is None:
        version = CURRENT_EXOME_RELEASE if data_type == 'exomes' else CURRENT_GENOME_RELEASE
    subdir = "sv/" if data_type == "genomes_snv_sv" else ""
    return f'gs://gnomad-public/release/{version}/ld/{subdir}gnomad.{data_type}.r{version}.{pop}.{"common." if common_only else ""}{"adj." if adj else ""}ld.bm'
